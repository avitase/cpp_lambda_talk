#include <algorithm>
#include <iostream>
#include <variant>
#include <vector>

template <typename... Fs> struct X: Fs... {
    using Fs::operator()...;
};
template <typename... Fs> X(Fs...) -> X<std::decay_t<Fs>...>;

int main() {
    int a = 0; double b = 0.;
    X visitor{[&a](int x) { a += x; },
              [&b](double x) { b += x; }};
    std::vector<std::variant<int, double>> v{1, 1.9, 2, 2.1};
    std::for_each(v.begin(), v.end(), [&visitor](const auto &x) {
        std::visit(visitor, x);
    });
    std::cout << a << ' ' << b;
}
