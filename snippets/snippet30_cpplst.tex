#include <algorithm>
#include <iostream>
#include <variant>
#include <vector>

struct X {
    int &a; double &b;
    auto operator()(int x) { a += x; };
    auto operator()(double x) { b += x; };
};

int main() {
    int a = 0; double b = 0.;
    X visitor{.a=a, .b=b};
    std::vector<std::variant<int, double>> v{1, 1.9, 2, 2.1};
    std::for_each(v.begin(), v.end(), [&visitor](const auto &x) {
        std::visit(visitor, x);
    });
    std::cout << a << b;
}
